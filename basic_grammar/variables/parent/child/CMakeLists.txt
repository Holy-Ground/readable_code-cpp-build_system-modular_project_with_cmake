set(CHILD_STRING_VALIABLE 1)

# PARENT_SCOPE: set the variable in the scope above the current scope
set(PARENT_STRING_VARIABLE 2 PARENT_SCOPE)

# CACHE: set the variable in the cache
# CACHE Tpye: BOOL, FILEPATH, PATH, STRING, INTERNAL, UNINITIALIZED
# BOOL: ON/OFF, TRUE/FALSE, YES/NO, 1/0
# FILEPATH: absolute path to a file
# PATH: absolute path to a directory
# STRING: any string
# INTERNAL: not shown in cmake-gui
# UNINITIALIZED: not initialized
set(GLOBAL_STRING_VARIABLE 3 CACHE STRING "a simple description for GLOBAL_VARIABLE_NAME")

# set(GLOBAL_UNINITIALIZED_VARIABLE 4 CACHE STRING "a simple description for GLOBAL_UNINITIALIZED_VARIABLE")
message(STATUS "GLOBAL_UNINITIALIZED_VARIABLE: ${GLOBAL_UNINITIALIZED_VARIABLE}\n")

message(STATUS "On child: ${CHILD_STRING_VALIABLE}")
message(STATUS "CHILD_STRING_VALIABLE: ${CHILD_STRING_VALIABLE}")
message(STATUS "PARENT_STRING_VARIABLE: ${PARENT_STRING_VARIABLE}")
message(STATUS "GLOBAL_STRING_VARIABLE: ${GLOBAL_STRING_VARIABLE}\n")
